all methods(GET,POST,etc) must include authentication key sa header, except sa login

user services:
    /login  --// login
    /user/{x} --// get specific user
    /user/{x} --// delete user
    /user/{x} --// update user

todo services:
    /user/{x}/todos --// get list of todos per user
    /user/{x}/todos --// create todos per user
    /user/{x}/todos/{x} --// get specific todo per user
    /user/{x}/todos/{x} --// update specific todo per user
    /user/{x}/todos/{x} --// delete specific todo per user

admin services:
    /todos  --//get list of todo
    /todos  --//delete all todo
    /todos/{x}  --//delete specific todo
    /users  --//get list of users
    /users  --//delete all users
    /users/{x}  --//update specific user
    /users/{x} --//delete specific user
